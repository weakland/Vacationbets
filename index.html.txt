<script>

const supabase = window.supabase.createClient(
"https://vzeagflemumbwmtyckea.supabase.co",
"sb_publishable_7Ze8piVXvFxvgEQTZhI_KA_uN18BATY"
)

const b=20

function price(yes,no){

let ey=Math.exp(yes/b)
let en=Math.exp(no/b)

return ey/(ey+en)

}

async function createMarket(){

let q=document.getElementById("question").value

await supabase
.from("markets")
.insert([{question:q}])

document.getElementById("question").value=""

loadMarkets()

}

async function buyYes(id){

await supabase.rpc("buy_yes",{market_id:id})

loadMarkets()

}

async function buyNo(id){

await supabase.rpc("buy_no",{market_id:id})

loadMarkets()

}

async function loadMarkets(){

let {data}=await supabase
.from("markets")
.select("*")

render(data)

}

async function loadChart(marketId,canvasId){

let {data}=await supabase
.from("price_history")
.select("*")
.eq("market_id",marketId)
.order("created_at",{ascending:true})

let prices=data.map(d=>d.price*100)

let labels=data.map((_,i)=>i+1)

new Chart(
document.getElementById(canvasId),
{
type:'line',
data:{
labels:labels,
datasets:[{
label:"YES Price %",
data:prices,
borderColor:"lime",
fill:false
}]
},
options:{
scales:{
y:{
min:0,
max:100
}
}
}
})

}

function render(markets){

let div=document.getElementById("markets")

div.innerHTML=""

markets.forEach(m=>{

let pYes=price(m.yes_shares,m.no_shares)
let pNo=1-pYes

let canvasId="chart"+m.id

div.innerHTML+=`

<div class="market">

<h3>${m.question}</h3>

<p>
YES ${(pYes*100).toFixed(1)}%
|
NO ${(pNo*100).toFixed(1)}%
</p>

<button class="yes" onclick="buyYes(${m.id})">Buy YES</button>
<button class="no" onclick="buyNo(${m.id})">Buy NO</button>

<canvas id="${canvasId}" height="120"></canvas>

</div>

`

setTimeout(()=>loadChart(m.id,canvasId),100)

})

}

loadMarkets()

setInterval(loadMarkets,3000)
