<script>

const supabase = window.supabase.createClient(
"YOUR_SUPABASE_URL",
"YOUR_SUPABASE_KEY"
)

const b=20

let currentUser=null

function price(yes,no){

let ey=Math.exp(yes/b)
let en=Math.exp(no/b)

return ey/(ey+en)

}

async function login(){

let name=document.getElementById("username").value

currentUser=name

let {data}=await supabase
.from("users")
.select("*")
.eq("name",name)

if(data.length===0){

await supabase
.from("users")
.insert([{name:name}])

}

loadWallet()

}

async function loadWallet(){

let {data}=await supabase
.from("users")
.select("*")
.eq("name",currentUser)
.single()

document.getElementById("balance").innerText=
currentUser+" balance: $"+data.balance.toFixed(2)

}

async function createMarket(){

let q=document.getElementById("question").value

await supabase
.from("markets")
.insert([{question:q}])

document.getElementById("question").value=""

loadMarkets()

}

async function buyYes(m){

if(!currentUser) return alert("login first")

let cost=1

await supabase
.rpc("increment_yes",{market_id:m.id})

}

async function buyNo(m){

if(!currentUser) return alert("login first")

await supabase
.rpc("increment_no",{market_id:m.id})

}

async function loadMarkets(){

let {data}=await supabase
.from("markets")
.select("*")

render(data)

}

function render(markets){

let div=document.getElementById("markets")

div.innerHTML=""

markets.forEach(m=>{

let pYes=price(m.yes_shares,m.no_shares)

let pNo=1-pYes

div.innerHTML+=`

<div class="market">

<h3>${m.question}</h3>

<p>

YES ${(pYes*100).toFixed(1)}%

|

NO ${(pNo*100).toFixed(1)}%

</p>

<button class="yes" onclick='buyYes(${JSON.stringify(m)})'>Buy YES</button>

<button class="no" onclick='buyNo(${JSON.stringify(m)})'>Buy NO</button>

</div>

`

})

loadLeaderboard()

}

async function loadLeaderboard(){

let {data}=await supabase
.from("users")
.select("*")
.order("balance",{ascending:false})

let html=""

data.forEach(u=>{

html+=u.name+" — $"+u.balance.toFixed(2)+"<br>"

})

document.getElementById("leaders").innerHTML=html

}

loadMarkets()

setInterval(loadMarkets,2000)
